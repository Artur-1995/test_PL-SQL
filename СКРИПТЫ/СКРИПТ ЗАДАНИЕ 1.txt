Задание:
Подготовьте DDL-скрипты создания объектов для приведённой модели: создание таблиц,
первичных, уникальных, внешних ключей и т.д.

Скрипт:

CREATE TABLE CLIENTS (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    NAME VARCHAR(1000),
    PLACE_OF_BIRTH VARCHAR(1000),
    DATE_OF_BIRTH DATE,
    ADDRESS VARCHAR(1000),
    PASSPORT VARCHAR(100)
) COMMENT '';

CREATE TABLE TARIFS (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    NAME VARCHAR(100),
    COST INT(10.2)
) COMMENT '';

CREATE TABLE PRODUCT_TYPES (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    NAME VARCHAR(100),
    BEGIN_DATE DATE,
    END_DATE DATE,
    TARIF_REF INT(10),
    CONSTRAINT PROD_TYPE_TAR_FK FOREIGN KEY (TARIF_REF) REFERENCES TARIFS (ID)
) COMMENT '';

CREATE TABLE PRODUCTS (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    PRODUCT_TYPE_ID INT(10),
    NAME VARCHAR(100),
    CLIENT_REF INT(10),
    OPEN_DATE DATE,
    CLOSE_DATE DATE,
    CONSTRAINT PROD_PRODTYPE_FK FOREIGN KEY (PRODUCT_TYPE_ID) REFERENCES PRODUCT_TYPES (ID),
    CONSTRAINT PROD_CL_FK FOREIGN KEY (CLIENT_REF) REFERENCES CLIENTS (ID)
) COMMENT '';

CREATE TABLE ACCOUNTS (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    NAME VARCHAR(100),
    SALDO INT(10.2),
    CLIENT_REF INT(10),
    OPEN_DATE DATE,
    CLOSE_DATE DATE,
    PRODUCT_REF INT(10),
    ACC_NUM VARCHAR(25),
    CONSTRAINT ACC_CL_FK FOREIGN KEY (CLIENT_REF) REFERENCES CLIENTS (ID),
    CONSTRAINT ACC_PROD_FK FOREIGN KEY (PRODUCT_REF) REFERENCES PRODUCTS (ID)
) COMMENT '';

CREATE TABLE RECORDS (
    ID INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    DT INT(1),
    SUM INT(10.2),
    ACC_REF INT(10),
    OPER_DATE DATE,
    CONSTRAINT REC_ACC_FK FOREIGN KEY (ACC_REF) REFERENCES ACCOUNTS (ID)
) COMMENT '';

commit;
